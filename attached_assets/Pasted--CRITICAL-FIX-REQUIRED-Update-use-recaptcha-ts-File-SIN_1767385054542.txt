```typescript
import { useEffect, useState, useCallback } from 'react';

declare global {
  interface Window {
    grecaptcha: {
      ready: (callback: () => void) => void;
      execute: (siteKey: string, options: { action:  string }) => Promise<string>;
    };
  }
}

const MAX_WAIT_TIME = 5000;

export function useRecaptcha() {
  const [isReady, setIsReady] = useState(false);
  const [loadFailed, setLoadFailed] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [siteKey, setSiteKey] = useState<string>('');

  useEffect(() => {
    const fetchSiteKey = async () => {
      try {
        const response = await fetch('/api/config/recaptcha');
        const data = await response.json();
        
        if (! data.siteKey) {
          console.error('reCAPTCHA site key not configured on server');
          setLoadFailed(true);
          setError('reCAPTCHA site key not configured');
          return;
        }
        
        setSiteKey(data.siteKey);
        console.log('✅ reCAPTCHA site key loaded from backend');
      } catch (err) {
        console.error('Failed to fetch reCAPTCHA config:', err);
        setLoadFailed(true);
        setError('Failed to load security configuration');
      }
    };

    fetchSiteKey();
  }, []);

  useEffect(() => {
    if (!siteKey) return;

    const startTime = Date.now();
    
    const checkRecaptcha = () => {
      if (window. grecaptcha) {
        try {
          window.grecaptcha.ready(() => {
            setIsReady(true);
            console.log('✅ reCAPTCHA ready');
          });
        } catch (err) {
          console. error('reCAPTCHA ready() failed:', err);
          setLoadFailed(true);
          setError('Security script initialization failed');
        }
      } else if (Date.now() - startTime > MAX_WAIT_TIME) {
        console.warn('⚠️ reCAPTCHA script failed to load after timeout');
        setLoadFailed(true);
        setError('Security script could not load');
      } else {
        setTimeout(checkRecaptcha, 200);
      }
    };

    checkRecaptcha();
  }, [siteKey]);

  const executeRecaptcha = useCallback(async (action: string): Promise<string | null> => {
    if (loadFailed) {
      return 'LOAD_FAILED';
    }
    
    if (!isReady || !siteKey) {
      console.warn('⚠️ reCAPTCHA not ready or not configured');
      return null;
    }

    try {
      const token = await window.grecaptcha. execute(siteKey, { action });
      console.log('✅ reCAPTCHA token generated');
      return token;
    } catch (err: unknown) {
      console.error('❌ reCAPTCHA execution failed:', err);
      return 'EXEC_FAILED';
    }
  }, [isReady, loadFailed, siteKey]);

  return { isReady: isReady || loadFailed, error, executeRecaptcha, loadFailed };
}
```
